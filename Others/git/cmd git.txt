git log color : 
git config --global alias.lg "log --graph --pretty=format:'%Cred%h%Creset -%C(yellow)%d%Creset %s %Cgreen(%cr) %C(bold blue)<%an>%Creset' --abbrev-commit --date=relative"
git config --global alias.lgg "log --graph --abbrev-commit --decorate --format=format:'%C(bold blue)%h%C(reset) - %C(bold green)(%ar)%C(reset) %C(white)%s%C(reset) %C(dim white)- %an%C(reset)%C(bold yellow)%d%C(reset)' --all"
git config --global alias.lgc "log --graph --abbrev-commit --decorate --format=format:'%C(bold blue)%h%C(reset) - %C(bold cyan)%aD%C(reset) %C(bold green)(%ar)%C(reset)%C(bold yellow)%d%C(reset)%n''          %C(white)%s%C(reset) %C(dim white)- %an%C(reset)' --all"

diff logs local/remote : 
git log --pretty=format:"%h%x09%an%x09%ad%x09%s" master..origin/master
git log --pretty=format:"%h - %an, %ar : %s" master..origin/master

nbr commit par pers : git shortlog -s -n
nbr commits local : git rev-list --count master
nbr commits remote : git rev-list --count origin/master

delete remote branch :
git push origin --delete <branche_name>
